🗄️ 2. Migration Tool (src/lib/storage/migration.ts)

```typescript
// src/lib/storage/migration.ts
// Migration Tool: localStorage → IndexedDB

import { CosmosDB } from "./indexeddb";

export interface MigrationResult {
  success: boolean;
  migrated: number;
  errors: number;
  logs: string[];
}

export class MigrationTool {
  private db: CosmosDB;

  constructor(db: CosmosDB) {
    this.db = db;
  }

  async migrate(): Promise<MigrationResult> {
    const logs: string[] = [];
    let migrated = 0;
    let errors = 0;

    try {
      // 1. ดึงข้อมูลจาก localStorage
      const keys = Object.keys(localStorage);
      const cosmosKeys = keys.filter(k => k.startsWith("cosmos."));

      logs.push(`📦 Found ${cosmosKeys.length} cosmos keys in localStorage`);

      for (const key of cosmosKeys) {
        try {
          const raw = localStorage.getItem(key);
          if (!raw) continue;

          const data = JSON.parse(raw);
          const storeName = this.mapKeyToStore(key);

          if (Array.isArray(data)) {
            for (const item of data) {
              await this.db.saveObject(storeName, item);
              migrated++;
            }
          } else {
            await this.db.saveObject(storeName, { id: key, ...data });
            migrated++;
          }

          logs.push(`✅ Migrated ${key} → ${storeName} (${Array.isArray(data) ? data.length : 1} items)`);
        } catch (err) {
          errors++;
          logs.push(`❌ Failed to migrate ${key}: ${String(err)}`);
        }
      }

      // 2. ตรวจสอบความถูกต้อง
      const verification = await this.verifyMigration();
      logs.push(`🔍 Verification: ${verification}`);

      // 3. ถ้าสำเร็จ → ลบ localStorage (optional)
      if (errors === 0) {
        // อย่าลบ localStorage อัตโนมัติ — ให้ user ยืนยัน
        logs.push("⚠️ localStorage data still exists. Run clearLocalStorage() when ready.");
      }

      return { success: errors === 0, migrated, errors, logs };

    } catch (err) {
      logs.push(`💥 Migration failed: ${String(err)}`);
      return { success: false, migrated, errors: errors + 1, logs };
    }
  }

  private mapKeyToStore(key: string): string {
    const map: Record<string, string> = {
      "cosmos.objects.v1": "objects",
      "cosmos.relationships.v1": "edges",
      "cosmos.collections.v1": "collections",
      "cosmos.events.v1": "events",
      "cosmos.soul.v1": "soulCards",
      "cosmos.settings.v1": "settings",
    };
    return map[key] || "objects";
  }

  async verifyMigration(): Promise<string> {
    try {
      const stores = ["objects", "edges", "collections", "events", "soulCards"];
      let total = 0;
      for (const store of stores) {
        const items = await this.db.getAllObjects(store);
        total += items.length;
      }
      return `✅ Verified ${total} items across ${stores.length} stores`;
    } catch (err) {
      return `❌ Verification failed: ${String(err)}`;
    }
  }

  async clearLocalStorage(): Promise<void> {
    const keys = Object.keys(localStorage).filter(k => k.startsWith("cosmos."));
    for (const key of keys) {
      localStorage.removeItem(key);
    }
  }
}
```

🎨 3. UI — Command Center (src/components/CommandCenter.tsx)

```tsx
// src/components/CommandCenter.tsx
// COSMOS Command Center — UI

import { useState, useEffect } from "react";
import { makaSync } from "../lib/ai/maka";
import { UniverseEngine } from "../lib/universe/engine";
import { GalaxyEngine } from "../lib/galaxy/engine";
import { PlanetEngine } from "../lib/planet/engine";
import { CosmosDB } from "../lib/storage/indexeddb";
import { EventEngine } from "../lib/events/engine";
import { QueryEngine } from "../lib/query/engine";

interface CommandCenterProps {
  onUniverseChange?: (universeId: string) => void;
}

export function CommandCenter({ onUniverseChange }: CommandCenterProps) {
  const [prompt, setPrompt] = useState("");
  const [response, setResponse] = useState("");
  const [loading, setLoading] = useState(false);
  const [universe, setUniverse] = useState<any>(null);
  const [galaxies, setGalaxies] = useState<any[]>([]);
  const [planets, setPlanets] = useState<any[]>([]);
  const [activeTab, setActiveTab] = useState<"maka" | "universe" | "galaxy" | "planet">("maka");

  useEffect(() => {
    const load = async () => {
      const db = new CosmosDB();
      await db.init();
      const eventEngine = new EventEngine(db);
      const queryEngine = new QueryEngine(db, null as any);

      const universeEngine = new UniverseEngine(db, eventEngine);
      await universeEngine.init();
      const active = await universeEngine.getActive();
      setUniverse(active);

      const galaxyEngine = new GalaxyEngine(db, eventEngine, queryEngine);
      setGalaxies(await galaxyEngine.getAll(active?.id));

      const planetEngine = new PlanetEngine(db, eventEngine, queryEngine, null as any);
      setPlanets(await planetEngine.getAll());
    };
    load();
  }, []);

  const handleMaka = () => {
    if (!prompt.trim()) return;
    setLoading(true);
    try {
      const result = makaSync(prompt);
      setResponse(result.output);
    } catch (err) {
      setResponse("❌ Error: " + String(err));
    }
    setLoading(false);
  };

  return (
    <div className="min-h-screen bg-background text-foreground p-6">
      <div className="max-w-6xl mx-auto">
        {/* Header */}
        <div className="flex items-center justify-between mb-6">
          <div>
            <h1 className="text-4xl font-bold text-gradient">🚀 COSMOS</h1>
            <p className="text-sm text-muted-foreground">
              Command Center · {universe?.name || "Loading..."}
            </p>
          </div>
          <div className="flex items-center gap-2">
            <span className="text-sm text-muted-foreground">🌌 {galaxies.length} Galaxies</span>
            <span className="text-sm text-muted-foreground">🪐 {planets.length} Planets</span>
          </div>
        </div>

        {/* Tabs */}
        <div className="flex gap-1 mb-6 glass p-1 rounded-2xl">
          {[
            { id: "maka", label: "🔥 Maka Prompt" },
            { id: "universe", label: "🌌 Universe" },
            { id: "galaxy", label: "🌠 Galaxies" },
            { id: "planet", label: "🪐 Planets" },
          ].map((tab) => (
            <button
              key={tab.id}
              onClick={() => setActiveTab(tab.id as any)}
              className={`px-4 py-2 rounded-xl text-sm transition-all ${
                activeTab === tab.id
                  ? "bg-primary/20 text-foreground"
                  : "text-muted-foreground hover:text-foreground"
              }`}
            >
              {tab.label}
            </button>
          ))}
        </div>

        {/* Tab Content */}
        <div className="glass rounded-3xl p-6">
          {/* Maka Prompt */}
          {activeTab === "maka" && (
            <div>
              <h2 className="text-xl font-semibold mb-3">🔥 Maka Prompt</h2>
              <p className="text-sm text-muted-foreground mb-4">
                พิมพ์อะไรก็ได้... AI จะเลือก Agent ที่เหมาะสมให้
              </p>
              <textarea
                className="w-full p-3 rounded-xl border border-border/40 bg-background/40 resize-none"
                rows={3}
                placeholder="พิมพ์ข้อความที่นี่..."
                value={prompt}
                onChange={(e) => setPrompt(e.target.value)}
                onKeyDown={(e) => e.key === "Enter" && !e.shiftKey && handleMaka()}
              />
              <button
                className="mt-3 px-6 py-2 rounded-full bg-gradient-to-r from-primary to-accent text-white font-medium disabled:opacity-50"
                onClick={handleMaka}
                disabled={loading || !prompt.trim()}
              >
                {loading ? "⏳ Processing..." : "🚀 Send"}
              </button>
              {response && (
                <div className="mt-4 p-4 rounded-xl bg-background/40 border border-border/40">
                  <p className="text-sm whitespace-pre-wrap">{response}</p>
                </div>
              )}
            </div>
          )}

          {/* Universe */}
          {activeTab === "universe" && (
            <div>
              <h2 className="text-xl font-semibold mb-3">🌌 Universe</h2>
              <div className="grid grid-cols-2 gap-4">
                <div className="p-4 rounded-xl bg-background/40 border border-border/40">
                  <p className="text-sm text-muted-foreground">Active</p>
                  <p className="text-2xl font-bold">{universe?.name || "—"}</p>
                  <p className="text-xs text-muted-foreground">{universe?.description}</p>
                </div>
                <div className="p-4 rounded-xl bg-background/40 border border-border/40">
                  <p className="text-sm text-muted-foreground">Status</p>
                  <p className="text-2xl font-bold">{universe?.status || "—"}</p>
                  <p className="text-xs text-muted-foreground">Mood: {universe?.mood}</p>
                </div>
              </div>
            </div>
          )}

          {/* Galaxies */}
          {activeTab === "galaxy" && (
            <div>
              <h2 className="text-xl font-semibold mb-3">🌠 Galaxies</h2>
              {galaxies.length === 0 ? (
                <p className="text-sm text-muted-foreground">No galaxies yet.</p>
              ) : (
                <div className="grid grid-cols-2 gap-3">
                  {galaxies.map((g) => (
                    <div key={g.id} className="p-3 rounded-xl bg-background/40 border border-border/40">
                      <div className="flex items-center gap-2">
                        <span className="text-xl">{g.icon}</span>
                        <span className="font-medium">{g.name}</span>
                      </div>
                      <p className="text-xs text-muted-foreground mt-1">{g.description}</p>
                    </div>
                  ))}
                </div>
              )}
            </div>
          )}

          {/* Planets */}
          {activeTab === "planet" && (
            <div>
              <h2 className="text-xl font-semibold mb-3">🪐 Planets</h2>
              {planets.length === 0 ? (
                <p className="text-sm text-muted-foreground">No planets yet.</p>
              ) : (
                <div className="grid grid-cols-2 gap-3">
                  {planets.map((p) => (
                    <div key={p.id} className="p-3 rounded-xl bg-background/40 border border-border/40">
                      <div className="flex items-center gap-2">
                        <span className="text-xl">{p.icon}</span>
                        <span className="font-medium">{p.name}</span>
                      </div>
                      <div className="flex gap-2 mt-1 text-xs text-muted-foreground">
                        <span>Lv.{p.level}</span>
                        <span>•</span>
                        <span>❤️ {p.health || 80}%</span>
                        <span>•</span>
                        <span className={p.status === "active" ? "text-green-400" : "text-muted-foreground"}>
                          {p.status}
                        </span>
                      </div>
                    </div>
                  ))}
                </div>
              )}
            </div>
          )}
        </div>

        {/* Footer */}
        <div className="mt-6 text-center text-xs text-muted-foreground">
          COSMOS-CAPTURE-OS v1.0 · Powered by Maka AI
        </div>
      </div>
    </div>
  );
}
```
