### 1.1 Context Builder (`src/lib/ai/context.ts`)

```typescript
// src/lib/ai/context.ts
// Context Builder — รวบรวมข้อมูลที่เกี่ยวข้องก่อนส่งให้ Agent

import { CosmosDB } from "../storage/indexeddb";
import { MemoryEngine } from "../memory/engine";
import { SoulEngine } from "../soul/engine";
import { GraphEngine } from "../graph/engine";
import { TimelineEngine } from "../timeline/engine";

export interface AgentContext {
  userIdentity: {
    name?: string;
    beliefs: string[];
    goals: string[];
    values: string[];
  };
  recentMemories: {
    id: string;
    title: string;
    content: string;
    emotionalScore: number;
  }[];
  relevantKnowledge: {
    id: string;
    title: string;
    content: string;
    confidence: number;
  }[];
  timeline: {
    recentActivities: string[];
    currentFocus: string;
    streak: number;
  };
  relationships: {
    connectedNodes: string[];
    strongestEdges: string[];
  };
  emotionalState: {
    mood: string;
    sentiment: "positive" | "negative" | "neutral";
    energy: number;
  };
}

export class ContextBuilder {
  constructor(
    private db: CosmosDB,
    private memoryEngine: MemoryEngine,
    private soulEngine: SoulEngine,
    private graphEngine: GraphEngine,
    private timelineEngine: TimelineEngine,
  ) {}

  async build(userId: string): Promise<AgentContext> {
    // 1. ดึง Soul (Identity)
    const souls = await this.soulEngine.getAll();
    const latestSoul = souls.length > 0 ? souls.reduce((a, b) => a.version > b.version ? a : b) : null;

    // 2. ดึง Memories ล่าสุด
    const memories = await this.memoryEngine.recall({ limit: 5, orderBy: "createdAt" });

    // 3. ดึง Knowledge (Graph)
    const graphData = this.graphEngine.exportGraph();
    const topNodes = graphData.nodes
      .sort((a, b) => (b.properties?.importance as number || 0) - (a.properties?.importance as number || 0))
      .slice(0, 5);

    // 4. ดึง Timeline
    const timeline = await this.timelineEngine.getTimeline({ granularity: "day", limit: 7 });

    // 5. คำนวณ Emotional State
    const recentEmotions = memories.filter(m => m.emotionalScore !== undefined).slice(0, 5);
    const avgEmotion = recentEmotions.length > 0
      ? recentEmotions.reduce((s, m) => s + m.emotionalScore, 0) / recentEmotions.length
      : 50;

    const sentiment: "positive" | "negative" | "neutral" = avgEmotion >= 65 ? "positive" : avgEmotion <= 35 ? "negative" : "neutral";

    return {
      userIdentity: {
        name: latestSoul?.name || "User",
        beliefs: latestSoul?.beliefs?.core || [],
        goals: latestSoul?.goals?.active || [],
        values: latestSoul?.values || [],
      },
      recentMemories: memories.slice(0, 5).map(m => ({
        id: m.id,
        title: m.title || "Untitled",
        content: m.content.slice(0, 200),
        emotionalScore: m.emotionalScore || 50,
      })),
      relevantKnowledge: topNodes.map(n => ({
        id: n.id,
        title: n.label || "Unknown",
        content: (n.properties?.content as string)?.slice(0, 200) || "",
        confidence: (n.properties?.confidence as number) || 50,
      })),
      timeline: {
        recentActivities: timeline.flatMap(p => p.entries.map(e => e.title || e.kind)).slice(0, 10),
        currentFocus: timeline.length > 0 ? timeline[timeline.length - 1].entries[0]?.title || "General" : "General",
        streak: 0, // TODO: จาก ReviewEngine
      },
      relationships: {
        connectedNodes: topNodes.map(n => n.id),
        strongestEdges: graphData.edges.sort((a, b) => b.weight - a.weight).slice(0, 3).map(e => e.id),
      },
      emotionalState: {
        mood: avgEmotion >= 65 ? "happy" : avgEmotion >= 45 ? "neutral" : "reflective",
        sentiment,
        energy: Math.min(100, avgEmotion * 1.2),
      },
    };
  }
}
```

```typescript
// src/lib/ai/intelligence.ts
// Real AI Integration — OpenRouter (ฟรี) + Gemini (ฟรี)

import { AgentContext } from "./context";

export type AIProvider = "openrouter" | "gemini" | "claude" | "mock";

export interface AIResponse {
  content: string;
  provider: AIProvider;
  confidence: number;
  tokensUsed?: number;
  reasoning?: string;
}

export class IntelligenceEngine {
  private apiKey: string;
  private provider: AIProvider;

  constructor(apiKey: string = "", provider: AIProvider = "mock") {
    this.apiKey = apiKey;
    this.provider = provider;
  }

  async callAI(
    prompt: string,
    context: AgentContext,
    systemPrompt?: string,
  ): Promise<AIResponse> {
    // ถ้าไม่มี API Key → ใช้ Mock
    if (!this.apiKey || this.provider === "mock") {
      return this.mockResponse(prompt, context);
    }

    // ถ้ามี API Key → ใช้ OpenRouter (ฟรี)
    if (this.provider === "openrouter") {
      return this.callOpenRouter(prompt, context, systemPrompt);
    }

    // ถ้าใช้ Gemini (ฟรี)
    if (this.provider === "gemini") {
      return this.callGemini(prompt, context, systemPrompt);
    }

    // ถ้าใช้ Claude (เสียเงิน)
    if (this.provider === "claude") {
      return this.callClaude(prompt, context, systemPrompt);
    }

    return this.mockResponse(prompt, context);
  }

  private async callOpenRouter(
    prompt: string,
    context: AgentContext,
    systemPrompt?: string,
  ): Promise<AIResponse> {
    const contextStr = this.formatContext(context);
    const fullPrompt = `${contextStr}\n\nUser: ${prompt}`;

    try {
      const response = await fetch("https://openrouter.ai/api/v1/chat/completions", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "Authorization": `Bearer ${this.apiKey}`,
          "HTTP-Referer": "https://cosmos-capture-os.com",
          "X-Title": "COSMOS-CAPTURE-OS",
        },
        body: JSON.stringify({
          model: "google/gemini-2.0-flash-001", // ฟรี!
          messages: [
            { role: "system", content: systemPrompt || "You are COSMOS, a wise and empathetic AI assistant." },
            { role: "user", content: fullPrompt },
          ],
          max_tokens: 1000,
          temperature: 0.7,
        }),
      });

      if (!response.ok) throw new Error(`HTTP ${response.status}`);
      const data = await response.json();
      const content = data.choices?.[0]?.message?.content || "No response";

      return {
        content,
        provider: "openrouter",
        confidence: 85,
        tokensUsed: data.usage?.total_tokens,
      };
    } catch (err) {
      console.error("[AI] OpenRouter failed:", err);
      return this.mockResponse(prompt, context);
    }
  }

  private async callGemini(
    prompt: string,
    context: AgentContext,
    systemPrompt?: string,
  ): Promise<AIResponse> {
    const contextStr = this.formatContext(context);
    const fullPrompt = `${systemPrompt || "You are COSMOS, a wise assistant."}\n\nContext:\n${contextStr}\n\nUser: ${prompt}`;

    try {
      const response = await fetch(
        `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${this.apiKey}`,
        {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({
            contents: [{ parts: [{ text: fullPrompt }] }],
            generationConfig: { maxOutputTokens: 1000, temperature: 0.7 },
          }),
        },
      );

      if (!response.ok) throw new Error(`HTTP ${response.status}`);
      const data = await response.json();
      const content = data.candidates?.[0]?.content?.parts?.[0]?.text || "No response";

      return {
        content,
        provider: "gemini",
        confidence: 85,
        tokensUsed: 0,
      };
    } catch (err) {
      console.error("[AI] Gemini failed:", err);
      return this.mockResponse(prompt, context);
    }
  }

  private async callClaude(
    prompt: string,
    context: AgentContext,
    systemPrompt?: string,
  ): Promise<AIResponse> {
    // TODO: Claude API integration (ต้องมี API Key เสียเงิน)
    return this.mockResponse(prompt, context);
  }

  private formatContext(context: AgentContext): string {
    const parts: string[] = [];

    if (context.userIdentity.name) {
      parts.push(`User: ${context.userIdentity.name}`);
    }
    if (context.userIdentity.beliefs.length) {
      parts.push(`Beliefs: ${context.userIdentity.beliefs.join(", ")}`);
    }
    if (context.userIdentity.goals.length) {
      parts.push(`Goals: ${context.userIdentity.goals.join(", ")}`);
    }
    if (context.recentMemories.length) {
      parts.push(`Recent Memories: ${context.recentMemories.map(m => m.title).join(", ")}`);
    }
    if (context.relevantKnowledge.length) {
      parts.push(`Related Knowledge: ${context.relevantKnowledge.map(k => k.title).join(", ")}`);
    }
    if (context.emotionalState.mood) {
      parts.push(`Current Mood: ${context.emotionalState.mood}`);
    }
    if (context.timeline.currentFocus) {
      parts.push(`Current Focus: ${context.timeline.currentFocus}`);
    }

    return parts.join("\n");
  }

  private mockResponse(prompt: string, context: AgentContext): AIResponse {
    const agent = this.selectMockAgent(prompt);
    const responses: Record<string, string> = {
      luffy: `🏴‍☠️ Luffy says: "${prompt}" — That's your dream! Let's make it happen. Stay bold, captain!`,
      zoro: `⚔️ Zoro says: "${prompt}" — I'll cut through this. Here's the execution plan.`,
      robin: `📚 Robin says: "${prompt}" — I've researched this based on your memories. Here are key insights.`,
      nami: `🗺️ Nami says: "${prompt}" — Let me calculate the resources. This will take ${Math.floor(Math.random() * 100) + 10} units.`,
      franky: `🔧 Franky says: "${prompt}" — SUPER! I'll build it. Here's the architecture.`,
      chopper: `🩺 Chopper says: "${prompt}" — Based on your current mood (${context.emotionalState.mood}), I recommend ${["walking", "sleep", "meditation", "exercise"][Math.floor(Math.random() * 4)]}.`,
      brook: `🎵 Brook says: "${prompt}" — Ah, I remember a story from your timeline...`,
      jinbe: `⚓ Jinbe says: "${prompt}" — I'll steady the ship. Here's your optimized routine.`,
      sanji: `🍳 Sanji says: "${prompt}" — Let me review this. I found ${Math.floor(Math.random() * 3) + 1} improvements.`,

    return {
      content: responses[agent] || `🤖 COSMOS says: "${prompt}" — I'll think about this and get back to you.`,
      provider: "mock",
      confidence: 60,
      reasoning: "Using mock response (no API key provided)",
    };
  }

  private selectMockAgent(input: string): string {
    const lower = input.toLowerCase();
    const agents = [
      { name: "luffy", keywords: ["dream", "goal", "vision", "purpose", "why"] },
      { name: "zoro", keywords: ["code", "build", "debug", "execute", "logic"] },
```

```typescript
// src/lib/ai/orchestrator.ts — อัปเดต (เชื่อม AI จริง)

import { IntelligenceEngine, AIResponse } from "./intelligence";
import { ContextBuilder, AgentContext } from "./context";
import { CosmosDB } from "../storage/indexeddb";
import { MemoryEngine } from "../memory/engine";
import { SoulEngine } from "../soul/engine";
import { GraphEngine } from "../graph/engine";
import { TimelineEngine } from "../timeline/engine";

export interface OrchestratorRequest {
  input: string;
  userId?: string;
  context?: Partial<AgentContext>;
  aiProvider?: "openrouter" | "gemini" | "claude" | "mock";
  apiKey?: string;
  mode?: "single" | "collaborative";
}

export interface OrchestratorResponse {
  primaryAgent: string;
  selectedAgents: string[];
  output: string;
  confidence: number;
  reasoning?: string;
  suggestions?: string[];
  usedContext?: boolean;
  aiProvider?: string;
}

export class Orchestrator {
  private contextBuilder: ContextBuilder;
  private intelligenceEngine: IntelligenceEngine;

  constructor(
    private db: CosmosDB,
    private memoryEngine: MemoryEngine,
    private soulEngine: SoulEngine,
    private graphEngine: GraphEngine,
    private timelineEngine: TimelineEngine,
  ) {
    this.contextBuilder = new ContextBuilder(db, memoryEngine, soulEngine, graphEngine, timelineEngine);
    this.intelligenceEngine = new IntelligenceEngine("", "mock");
  }

  async orchestrate(request: OrchestratorRequest): Promise<OrchestratorResponse> {
    const { input, userId, apiKey, aiProvider = "mock", mode = "single" } = request;

    // 1. Build Context (ถ้ามี userId)
    let context: AgentContext | null = null;
    if (userId) {
      context = await this.contextBuilder.build(userId);
    }

    // 2. Initialize AI Engine
    this.intelligenceEngine = new IntelligenceEngine(apiKey || "", aiProvider);

    // 3. Select Agent
    const agent = this.selectAgent(input, context);

    // 4. Build System Prompt
    const systemPrompt = this.buildSystemPrompt(agent, context);

    // 5. Call AI
    const aiResponse = await this.intelligenceEngine.callAI(input, context || this.getDefaultContext(), systemPrompt);

    // 6. Generate Suggestions
    const suggestions = this.generateSuggestions(input, agent, context);

    return {
      primaryAgent: agent,
      selectedAgents: [agent],
      output: aiResponse.content,
      confidence: aiResponse.confidence,
      reasoning: aiResponse.reasoning,
      suggestions,
      usedContext: !!context,
      aiProvider: aiProvider,
    };
  }

  private selectAgent(input: string, context: AgentContext | null): string {
    // Agent Selection Logic (ใช้ทั้ง keyword + context)
    const lower = input.toLowerCase();
    const agents = [
      { name: "luffy", keywords: ["dream", "goal", "vision", "purpose", "why"], contextMatch: ["goals"] },
      { name: "zoro", keywords: ["code", "build", "debug", "execute", "logic"], contextMatch: ["recent_activities"] },
      { name: "robin", keywords: ["research", "knowledge", "learn", "history", "insight"], contextMatch: ["knowledge"] },
      { name: "nami", keywords: ["finance", "budget", "plan", "resource", "cost"], contextMatch: ["goals"] },
      { name: "chopper", keywords: ["health", "sleep", "exercise", "mental", "stress"], contextMatch: ["emotional_state"] },
      { name: "brook", keywords: ["memory", "story", "emotion", "nostalgia", "past"], contextMatch: ["memories"] },
      { name: "jinbe", keywords: ["routine", "discipline", "process", "daily", "workflow"], contextMatch: ["timeline"] },
    ];

    let bestScore = 0;
    let bestAgent = "jinbe";

    for (const agent of agents) {
      let score = 0;
      // Keyword match
      for (const keyword of agent.keywords) {
        if (lower.includes(keyword)) score += 2;
      }
      // Context match
      if (context) {
        for (const match of agent.contextMatch) {
          if (match === "goals" && context.userIdentity.goals.length) score += 1;
          if (match === "memories" && context.recentMemories.length) score += 1;
          if (match === "knowledge" && context.relevantKnowledge.length) score += 1;
          if (match === "emotional_state" && context.emotionalState.mood) score += 1;
          if (match === "timeline" && context.timeline.recentActivities.length) score += 1;
        }
      }
      if (score > bestScore) { bestScore = score; bestAgent = agent.name; }
    }

    return bestScore > 0 ? bestAgent : "jinbe";
  }

  private buildSystemPrompt(agent: string, context: AgentContext | null): string {
    const agentPrompts: Record<string, string> = {
      luffy: "You are Luffy — Vision AI of COSMOS. Inspire, think big, cut through overthinking. Focus on dreams and purpose.",
      zoro: "You are Zoro — Execution AI. Write clean code, solve problems step-by-step. Be concise. No fluff.",
      robin: "You are Robin — Knowledge AI. Deep research, connect dots, surface hidden insights. Reference past data.",
      nami: "You are Nami — Resource AI. Track resources, manage budgets, create plans. Precise with numbers.",
      chopper: "You are Chopper — Health AI. Monitor wellbeing, give actionable advice. Caring but evidence-based.",
      brook: "You are Brook — Memory AI. Preserve memories, surface nostalgia, connect emotionally with the past.",
      jinbe: "You are Jinbe — Operations AI. Maintain stability, enforce discipline, optimize daily workflows.",
      franky: "You are Franky — Builder AI. Design systems, tools, infrastructure. Think in components.",
      sanji: "You are Sanji — Quality AI. Review code, audit security, polish output. Perfectionist but constructive.",
```

```typescript
// src/lib/ai/learning.ts
// Self-Learning Loop — เรียนรู้จากพฤติกรรมผู้ใช้และปรับปรุงตัวเอง

import { CosmosDB } from "../storage/indexeddb";
import { EventEngine } from "../events/engine";
import { QueryEngine } from "../query/engine";

export interface UserFeedback {
  id: string;
  timestamp: number;
  input: string;
  response: string;
  rating: number; // 1-5
  liked: boolean;
  agent: string;
  tags: string[];
  improved: boolean;
}

export class LearningEngine {
  private feedbacks: UserFeedback[] = [];

  constructor(
    private db: CosmosDB,
    private eventEngine: EventEngine,
    private queryEngine: QueryEngine,
  ) {}

  async recordFeedback(feedback: Omit<UserFeedback, "id" | "timestamp">): Promise<UserFeedback> {
    const entry: UserFeedback = {
      id: crypto.randomUUID(),
      timestamp: Date.now(),
      ...feedback,
    };
    this.feedbacks.push(entry);
    await this.db.saveObject("objects", { ...entry, objectType: "feedback" });
    return entry;
  }

  async getFeedbacks(since?: number): Promise<UserFeedback[]> {
    const result = await this.queryEngine.query<UserFeedback>({
      type: "feedback",
      where: since ? { field: "timestamp", operator: "gte", value: since } : undefined,
      options: { orderBy: "timestamp", orderDir: "desc", limit: 100 },
    });
    return result.data;
  }

  async analyzePatterns(): Promise<{
    topAgents: { agent: string; count: number; avgRating: number }[];
    commonTopics: { tag: string; count: number }[];
    improvementAreas: string[];
  }> {
    const feedbacks = await this.getFeedbacks();

    // Agent Performance
    const agentMap = new Map<string, { count: number; totalRating: number }>();
    const tagMap = new Map<string, number>();
    const improvements: string[] = [];

    for (const f of feedbacks) {
      if (!agentMap.has(f.agent)) {
        agentMap.set(f.agent, { count: 0, totalRating: 0 });
      }
      const agentData = agentMap.get(f.agent)!;
      agentData.count++;
      agentData.totalRating += f.rating;

      for (const tag of f.tags) {
        tagMap.set(tag, (tagMap.get(tag) || 0) + 1);
      }

      if (!f.improved && f.rating < 3) {
        improvements.push(`${f.agent} improvement needed for: "${f.input.slice(0, 50)}"`);
      }
    }

    const topAgents = Array.from(agentMap.entries())
      .map(([agent, data]) => ({
        agent,
        count: data.count,
        avgRating: Math.round(data.totalRating / data.count),
      }))
      .sort((a, b) => b.avgRating - a.avgRating)
      .slice(0, 5);

    const commonTopics = Array.from(tagMap.entries())
      .map(([tag, count]) => ({ tag, count }))
      .sort((a, b) => b.count - a.count)
      .slice(0, 10);

    return { topAgents, commonTopics, improvementAreas: improvements.slice(0, 5) };
  }

  async improveAgent(agent: string, feedback: UserFeedback[]): Promise<string> {
    // TODO: ใช้ feedback เพื่อปรับปรุง agent prompt (ต้องการ AI API)
    const avgRating = feedback.reduce((s, f) => s + f.rating, 0) / feedback.length;
    if (avgRating < 3) {
      return `Agent ${agent} needs improvement (avg rating: ${avgRating.toFixed(1)}). Consider updating its system prompt.`;
    }
    return `Agent ${agent} is performing well (avg rating: ${avgRating.toFixed(1)}).`;
  }
}
```
