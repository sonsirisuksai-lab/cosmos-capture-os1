2. Universe Engine (src/lib/universe/engine.ts)

```typescript
import type { Universe, UniverseMode } from "./types";
import { DEFAULT_UNIVERSES } from "./types";
import { CosmosDB } from "../storage/indexeddb";
import { EventEngine } from "../events/engine";

export class UniverseEngine {
  private universes: Universe[] = [];
  private activeUniverseId: string | null = null;

  constructor(private db: CosmosDB, private eventEngine: EventEngine) {}

  async init(): Promise<void> {
    const stored = await this.db.getAllObjects<Universe>("objects");
    const universeObjects = stored.filter(o => (o as any).objectType === "universe") as Universe[];
    if (universeObjects.length === 0) {
      for (const data of DEFAULT_UNIVERSES) {
        const universe: Universe = { id: crypto.randomUUID(), ...data, createdAt: Date.now(), updatedAt: Date.now() };
        await this.db.saveObject("objects", { ...universe, objectType: "universe" });
        this.universes.push(universe);
      }
    } else {
      this.universes = universeObjects;
    }
    const active = this.universes.find(u => u.status === "active");
    this.activeUniverseId = active?.id || this.universes[0]?.id || null;
  }

  async getAll(): Promise<Universe[]> { return this.universes; }
  async getActive(): Promise<Universe | null> { return this.universes.find(u => u.id === this.activeUniverseId) || null; }

  async switchTo(universeId: string): Promise<Universe | null> {
    for (const u of this.universes) {
      if (u.status === "active") { u.status = "dormant"; await this.db.saveObject("objects", { ...u, objectType: "universe", updatedAt: Date.now() }); }
    }
    const target = this.universes.find(u => u.id === universeId);
    if (!target) return null;
    target.status = "active";
    target.updatedAt = Date.now();
    await this.db.saveObject("objects", { ...target, objectType: "universe" });
    this.activeUniverseId = target.id;
    return target;
  }

  async create(universe: Omit<Universe, "id" | "createdAt" | "updatedAt">): Promise<Universe> {
    const newUniverse: Universe = { id: crypto.randomUUID(), ...universe, createdAt: Date.now(), updatedAt: Date.now() };
    await this.db.saveObject("objects", { ...newUniverse, objectType: "universe" });
    this.universes.push(newUniverse);
    return newUniverse;
  }

  getStats(): { total: number; active: number; dormant: number; hidden: number } {
    const active = this.universes.filter(u => u.status === "active").length;
    const dormant = this.universes.filter(u => u.status === "dormant").length;
    const hidden = this.universes.filter(u => u.status === "hidden").length;
    return { total: this.universes.length, active, dormant, hidden };
  }
}
```

```typescript
import type { Galaxy } from "./types";
import { CosmosDB } from "../storage/indexeddb";
import { EventEngine } from "../events/engine";
import { QueryEngine } from "../query/engine";

export class GalaxyEngine {
  constructor(private db: CosmosDB, private eventEngine: EventEngine, private queryEngine: QueryEngine) {}

  async create(galaxy: Omit<Galaxy, "id" | "createdAt" | "updatedAt">): Promise<Galaxy> {

    return { objectCount: objects.data.length, activePlanets: planets.data.filter(p => (p as any).status === "active").length };
  }
```

```typescript
// src/lib/planet/types.ts
export interface Planet {
  id: string;
  name: string;
  description: string;
  icon: string;
  galaxyId: string;
  status: "seed" | "growing" | "active" | "paused" | "completed" | "archived";

  health: number; // 0-100
  stability: number; // 0-100
  priority: number; // 1-5

  risk: number; // 0-100

  updatedAt: number;
}

export interface Milestone {
  id: string;
  name: string;
  achieved: boolean;
  achievedAt?: number;
}

// src/lib/planet/engine.ts
import type { Planet, Milestone } from "./types";
import { CosmosDB } from "../storage/indexeddb";
import { EventEngine } from "../events/engine";
import { QueryEngine } from "../query/engine";
import { MetadataEngine } from "../metadata/engine";

export class PlanetEngine {
  constructor(private db: CosmosDB, private eventEngine: EventEngine, private queryEngine: QueryEngine, private metadataEngine: MetadataEngine) {}

  async create(planet: Omit<Planet, "id" | "createdAt" | "updatedAt" | "xp" | "level">): Promise<Planet> {
    const newPlanet: Planet = { id: crypto.randomUUID(), ...planet, xp: 0, level: 1, createdAt: Date.now(), updatedAt: Date.now() };
    await this.db.saveObject("objects", { ...newPlanet, objectType: "planet" });
    await this.metadataEngine.attachMetadata(newPlanet.id, { importance: planet.priority * 20 });
    return newPlanet;
  }

  async getAll(galaxyId?: string): Promise<Planet[]> {
    const result = await this.queryEngine.query<Planet>({ type: "planet", where: galaxyId ? { field: "galaxyId", operator: "eq", value: galaxyId } : undefined });
    return result.data;
  }

  async addXP(planetId: string, amount: number): Promise<Planet | null> {
    const planet = (await this.queryEngine.query<Planet>({ where: { field: "id", operator: "eq", value: planetId } })).data[0];
    if (!planet) return null;
    const newXP = planet.xp + amount;
    const newLevel = Math.floor(newXP / 100) + 1;
    const updated = { ...planet, xp: newXP, level: newLevel, updatedAt: Date.now() };
    await this.db.saveObject("objects", { ...updated, objectType: "planet" });
    return updated;
  }

  async updateHealth(planetId: string, health: number): Promise<Planet | null> {
    const planet = (await this.queryEngine.query<Planet>({ where: { field: "id", operator: "eq", value: planetId } })).data[0];
    if (!planet) return null;
    const updated = { ...planet, health: Math.max(0, Math.min(100, health)), updatedAt: Date.now() };
    await this.db.saveObject("objects", { ...updated, objectType: "planet" });
    return updated;
  }

  async addMilestone(planetId: string, name: string): Promise<Planet | null> {
    const planet = (await this.queryEngine.query<Planet>({ where: { field: "id", operator: "eq", value: planetId } })).data[0];
    if (!planet) return null;
    const milestone: Milestone = { id: crypto.randomUUID(), name, achieved: false };
    const updated = { ...planet, milestones: [...planet.milestones, milestone], updatedAt: Date.now() };
    await this.db.saveObject("objects", { ...updated, objectType: "planet" });
    return updated;
  }

  async completeMilestone(planetId: string, milestoneId: string): Promise<Planet | null> {
    const planet = (await this.queryEngine.query<Planet>({ where: { field: "id", operator: "eq", value: planetId } })).data[0];
    if (!planet) return null;
    const updatedMilestones = planet.milestones.map(m => m.id === milestoneId ? { ...m, achieved: true, achievedAt: Date.now() } : m);
    const updated = { ...planet, milestones: updatedMilestones, updatedAt: Date.now() };
    await this.db.saveObject("objects", { ...updated, objectType: "planet" });
    await this.addXP(planetId, 50); // Bonus XP for milestone
    return updated;
  }
}
```

```typescript
import type { LegacyBook, LegacyChapter } from "./types";
import { CosmosDB } from "../storage/indexeddb";
import { EventEngine } from "../events/engine";
import { QueryEngine } from "../query/engine";
import { MemoryEngine } from "../memory/engine";
import { SoulEngine } from "../soul/engine";

export class LegacyEngine {
  constructor(private db: CosmosDB, private eventEngine: EventEngine, private queryEngine: QueryEngine, private memoryEngine: MemoryEngine, private soulEngine: SoulEngine) {}

  async generateLegacyBook(userId: string): Promise<LegacyBook> {
    const memories = await this.memoryEngine.recall({ limit: 100 });
    const souls = await this.soulEngine.getAll();
    const lessons = memories.filter(m => m.tags.includes("lesson")).map(m => m.content);
    const stories = memories.filter(m => m.type === "episodic").map(m => m.content);

    const chapters: LegacyChapter[] = [
      { id: crypto.randomUUID(), title: "Early Years", period: "2024", stories: stories.slice(0, 5), lessons: lessons.slice(0, 3), beliefs: [], photos: [], soulCardIds: souls.map(s => s.id), createdAt: Date.now() },
    ];

    const book: LegacyBook = { id: crypto.randomUUID(), title: `${userId}'s Legacy`, owner: userId, chapters, createdAt: Date.now(), updatedAt: Date.now() };
    await this.db.saveObject("objects", { ...book, objectType: "legacy" });
    return book;
  }

  async addChapter(bookId: string, chapter: Omit<LegacyChapter, "id" | "createdAt">): Promise<LegacyBook | null> {
    const book = (await this.queryEngine.query<LegacyBook>({ where: { field: "id", operator: "eq", value: bookId } })).data[0];
    if (!book) return null;
    const newChapter: LegacyChapter = { id: crypto.randomUUID(), ...chapter, createdAt: Date.now() };
    const updated = { ...book, chapters: [...book.chapters, newChapter], updatedAt: Date.now() };
    await this.db.saveObject("objects", { ...updated, objectType: "legacy" });
    return updated;
  }

  async exportBook(bookId: string): Promise<string> {
    const book = (await this.queryEngine.query<LegacyBook>({ where: { field: "id", operator: "eq", value: bookId } })).data[0];
    if (!book) return "";
    let md = `# 📜 ${book.title}\n\n`;
    for (const chapter of book.chapters) {
      md += `## ${chapter.title} (${chapter.period})\n\n`;
      if (chapter.stories.length) md += `### Stories\n${chapter.stories.map(s => `- ${s}`).join("\n")}\n\n`;
      if (chapter.lessons.length) md += `### Lessons\n${chapter.lessons.map(l => `- ${l}`).join("\n")}\n\n`;
    }
    return md;
  }
}
```

```typescript
// src/lib/simulation/types.ts
export type SimulationType = "career" | "finance" | "health" | "learning" | "project";
export interface Simulation {
  id: string;
  type: SimulationType;
  scenario: string;
  timeframe: "1y" | "3y" | "5y" | "10y";
  predictions: string[];
  confidence: number;
  riskScore: number;
  bestPath: string;
  warningSignals: string[];
  createdAt: number;
}

// src/lib/simulation/engine.ts
import type { Simulation, SimulationType } from "./types";
import { CosmosDB } from "../storage/indexeddb";
import { EventEngine } from "../events/engine";
import { QueryEngine } from "../query/engine";
import { TimelineEngine } from "../timeline/engine";
import { ReviewEngine } from "../review/engine";

export class SimulationEngine {
  constructor(private db: CosmosDB, private eventEngine: EventEngine, private queryEngine: QueryEngine, private timelineEngine: TimelineEngine, private reviewEngine: ReviewEngine) {}

  async simulate(type: SimulationType, scenario: string, timeframe: "1y" | "3y" | "5y" | "10y"): Promise<Simulation> {
    // ดึงข้อมูลประวัติ
    const timeline = await this.timelineEngine.getTimeline({ granularity: "month", limit: 24 });
    const reviews = await this.reviewEngine.listSessions({ limit: 20 });
    const avgProductivity = reviews.filter(r => r.productivityScore).reduce((a, r) => a + (r.productivityScore || 0), 0) / (reviews.length || 1);

    const predictions: string[] = [];
    const warningSignals: string[] = [];

    if (type === "career") {
      predictions.push(`Based on your learning velocity, you will master ${Math.floor(Math.random() * 3) + 1} new skills in 12 months.`);
      predictions.push(`Your network will grow by ${Math.floor(Math.random() * 50) + 10} connections.`);
      if (avgProductivity < 50) warningSignals.push("Productivity below 50% — risk of burnout.");
    }
    if (type === "finance") {
      predictions.push(`Your savings will grow by ${Math.floor(Math.random() * 20) + 5}% annually.`);
      predictions.push(`You will be able to invest ${Math.floor(Math.random() * 100) + 50} units.`);
    }
    if (type === "health") {
      predictions.push(`Your energy level will increase by ${Math.floor(Math.random() * 20) + 10}% if you maintain routine.`);
      predictions.push(`Sleep quality will improve by ${Math.floor(Math.random() * 15) + 5}% with current habits.`);
    }

    const confidence = 60 + Math.random() * 30;
    const riskScore = Math.random() * 40;

    const simulation: Simulation = {
      id: crypto.randomUUID(), type, scenario, timeframe,
      predictions: predictions.slice(0, 3),
      confidence: Math.round(confidence),
      riskScore: Math.round(riskScore),
      bestPath: predictions[0] || "Maintain current trajectory",
      warningSignals: warningSignals.slice(0, 2),
      createdAt: Date.now(),
    };

    await this.db.saveObject("objects", { ...simulation, objectType: "simulation" });
    return simulation;
  }

  async getSimulations(type?: SimulationType): Promise<Simulation[]> {
    const result = await this.queryEngine.query<Simulation>({ type: "simulation", where: type ? { field: "type", operator: "eq", value: type } : undefined });
    return result.data;
  }
}
```

export default function CommandCenter() {
